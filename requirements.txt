# Runtime dependencies for the PaddleOCR MCP server.
# Versions are pinned for reproducible Docker builds; bump deliberately.
# The heavy ML stack (paddleocr/paddlex/modelscope) and mcp both depend on
# pydantic; with loose ranges pip's resolver backtracks through dozens of
# pydantic/pydantic-core candidates and can fail with a spurious
# "ResolutionImpossible" on the slim base image. Pinning the shared libs to the
# known-good set below makes resolution deterministic and fast.
mcp==1.28.1
pydantic==2.13.4
pydantic-core==2.46.4
paddleocr[doc-parser]==3.7.0
# The `doc-parser` extra pulls in paddlex[ocr] (premailer, openpyxl, lxml,
# shapely, scikit-learn, scipy, latex2mathml, ...) which PP-StructureV3 needs
# for layout / table analysis. Plain `paddleocr` only brings paddlex[ocr-core],
# which is enough for text OCR but makes PP-StructureV3 fail with
# "A dependency error occurred during pipeline creation".
onnxruntime>=1.16
# Pin paddlepaddle to a version that ships pre-built wheels for BOTH
# linux/amd64 and linux/aarch64 (3.3.x has no aarch64 wheel). 3.2.2 is a
# drop-in since paddleocr/paddlex do not constrain paddlepaddle.
paddlepaddle==3.2.2
# Pin paddle2onnx to 1.3.1 to avoid onnxoptimizer. Newer paddle2onnx (>=2.x)
# depends on `onnxoptimizer`, which has NO linux/aarch64 wheel on PyPI and
# would force a slow, fragile cmake build from source. 1.3.1 has aarch64
# wheels and no such dependency. paddle2onnx is only used for paddle->onnx
# *conversion*; this project never imports it (inference uses pre-downloaded
# models directly), so the pinned version is purely transitive.
paddle2onnx==1.3.1
pillow>=10.0
numpy>=1.24,<3
anyio>=4.0
